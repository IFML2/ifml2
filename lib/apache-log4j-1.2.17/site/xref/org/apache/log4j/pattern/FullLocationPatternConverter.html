18  package org.apache.log4j.pattern;
19  
20  import org.apache.log4j.spi.LocationInfo;
21  import org.apache.log4j.spi.LoggingEvent;
22  
23  
24  /**
25   * Format the event's line location information.
26   *
27   * @author Ceki G&uuml;lc&uuml;
28   */
29  public final class FullLocationPatternConverter
30    extends LoggingEventPatternConverter {
31    /**
32     * Singleton.
33     */
34    private static final FullLocationPatternConverter INSTANCE =
35      new FullLocationPatternConverter();
36  
37    /**
38     * Private constructor.
39     */
40    private FullLocationPatternConverter() {
41      super("Full Location", "fullLocation");
42    }
43  
44    /**
45     * Obtains an instance of pattern converter.
46     * @param options options, may be null.
47     * @return instance of pattern converter.
48     */
49    public static FullLocationPatternConverter newInstance(
50      final String[] options) {
51      return INSTANCE;
52    }
53  
54    /**
55     * {@inheritDoc}
56     */
57    public void format(final LoggingEvent event, final StringBuffer output) {
58      LocationInfo locationInfo = event.getLocationInformation();
59  
60      if (locationInfo != null) {
61        output.append(locationInfo.fullInfo);
62      }
63    }
64  }
