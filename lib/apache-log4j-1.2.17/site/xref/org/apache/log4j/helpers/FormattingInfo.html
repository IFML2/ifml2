21  /**

23     formatting modifiers in conversion modifiers.
24  
25     @author <a href=mailto:jim_cakalic@na.biomerieux.com>Jim Cakalic</a>
26     @author Ceki G&uuml;lc&uuml;

30  public class FormattingInfo {
31    int min = -1;
32    int max = 0x7FFFFFFF;
33    boolean leftAlign = false;
34  
35    void reset() {
36      min = -1;
37      max = 0x7FFFFFFF;
38      leftAlign = false;      
39    }
40

43    }
44  }
