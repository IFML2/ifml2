31                                                                   UnexpectedFormatException {
32  
33      String line;
34      BufferedReader input = new BufferedReader(new FileReader(in));
35      PrintStream output = new PrintStream(new FileOutputStream(out, false));
36    
37      // Initialization of input and output omitted
38      while((line = input.readLine()) != null) {
39        // apply all filters
40        for(int i = 0; i < filters.length; i++) {
41  	line = filters[i].filter(line);
42        }
43        if(line != null) {
44  	output.println(line);
45        }
46      }
47    }
